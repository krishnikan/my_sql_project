-- 3. TECHNICAL OFFICER TABLE 

CREATE TABLE Technical_officer (
    To_id CHAR(10) PRIMARY KEY,
    User_id CHAR(10),
    Name VARCHAR(100),
    Gender CHAR(5),
    NIC CHAR(12),
    Address VARCHAR(100),
    Phone_no CHAR(10),
    FOREIGN KEY (User_id) REFERENCES Users(User_id)
);
